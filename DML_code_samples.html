    customer_promotion_id,
    customer_id,
    promotion_id,
    is_used,
    date_offered,
    expiry_date,
    times_used,
    usage_limit,
    CASE 
        WHEN expiry_date < SYSDATE THEN 'Expired'

        WHEN is_used = 1 THEN 'Used'

    p.discount_type,

-- Run query with calculated line item total for ORDER_ITEMS
SELECT 
    order_item_id,
    order_id,
    product_id,
    quantity,
    unit_price,
    discount,
    (quantity * unit_price) - discount AS item_total,
    created_at,
    updated_at
FROM ORDER_ITEMS
ORDER BY order_id, order_item_id;

-- Run query with formatted period for SUBSCRIPTION_ITEMS
SELECT 
    subscription_item_id,
    subscription_id,
    product_id,
    month,
    year,
    quantity,
    TO_CHAR(TO_DATE(month || '/01/' || year, 'MM/DD/YYYY'), 'Month YYYY') AS period,
    created_at,
    updated_at
FROM SUBSCRIPTION_ITEMS
ORDER BY subscription_id, year, month, subscription_item_id;
